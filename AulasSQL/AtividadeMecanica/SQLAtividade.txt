   -- JOIN
   
   SELECT os.*, cli.Nome, vei.Placa, os.Data FROM Ordem_servico os
   JOIN Veiculo vei ON os.Id_veiculo = vei.Id_veiculo
   JOIN Cliente cli ON vei.Id_cliente = cli.Id_cliente;
   
   SELECT po.*, prod.Nome FROM Produto_ordem po
   JOIN Produtos prod ON po.Id_produto = prod.Id_peca
   WHERE po.Id_os = 50;
   
   SELECT os.*, meca.Nome FROM Ordem_servico os
   JOIN Mecanico meca ON os.Id_mecanico = meca.Id_mecanico
   WHERE os.Id_os = 50;
